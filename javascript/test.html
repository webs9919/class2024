01. 1부터 100까지 출력하기(for문, while, do while)

for문을 이용해서 1-100까지 출력하는 예제입니다.

{
    for (let i = 0; i < 100; i++) {
        console.log(i + 1);
    }

    // 1,2,3,4 ....100
}
{
    let num = 0;    // 초깃값 설정
    while (num < 100) {
        console.log(num + 1);
        num++;
    }
}
{
    let num = 0;
    do {
        console.log(num + 1);
        num++;
    } while (num < 100)
}
